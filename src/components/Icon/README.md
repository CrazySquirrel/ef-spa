# Icon

```typescript
interface Icon {
  viewBox: string;
  id: string;
}
```

```typescript
interface IconProps {
  fill?: string;
  ref?: (node: SVGElement) => void;
}
```

## Usage

```typescript
import {<someIcon>} from 'components/Icon';
```

```typescript
{<someIcon>()}
```

import {IconChat} from 'components/Icon';
